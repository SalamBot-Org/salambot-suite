    geminiClient: GeminiClient,
    cache: RedisCache

    this.geminiClient = geminiClient;
    this.cache = cache;

  async detectLanguage(text: string): Promise<DarijaDetectionResult> {
    // Validation d'entr√©e
    if (!text?.trim()) {

    // D√©tection IA

    // Cache storage
    await this.cache.set(cacheKey, result, { ttl: 3600 });
